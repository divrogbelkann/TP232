        // Clé API JSONBin (gratuit - à remplacer par votre clé)

        document.getElementById('collectForm').addEventListener('submit', async (e) => {
            e.preventDefault();
            
            const newData = {
                id: Date.now(),
                plage: document.getElementById('plage').value,
                date: document.getElementById('date').value,
                quantite: parseFloat(document.getElementById('quantite').value),
                type: document.getElementById('type').value,
                benevoles: parseInt(document.getElementById('benevoles').value),
                timestamp: new Date().toISOString()
            };
            
            // Récupérer les données existantes
            let existingData = getData();
            existingData.push(newData);
            
            // Sauvegarder
            if (saveData(existingData)) {
                document.getElementById('message').style.display = 'block';
                document.getElementById('message').innerHTML = '✅ Donnée enregistrée avec succès !';

                // Réinitialiser le formulaire

                document.getElementById('date').valueAsDate = new Date();
                
                // Mettre à jour les affichages

            } else {

        // Fonctions de gestion des données
        function getData() {
            const stored = localStorage.getItem('ecoOceanData');
            if (stored) {
                return JSON.parse(stored);
            }
            return [];
        }
        
        function saveData(data) {
            try {
                localStorage.setItem('ecoOceanData', JSON.stringify(data));
                return true;
            } catch(e) {
                return false;
            }
        }
        
        // Statistiques descriptives
        function calculateStats(data) {
            if (data.length === 0) return null;
            
            const quantites = data.map(d => d.quantite).sort((a,b) => a-b);
            const sum = quantites.reduce((a,b) => a+b, 0);
            const mean = sum / data.length;
            
            // Médiane
            let median;
            const mid = Math.floor(quantites.length / 2);
            if (quantites.length % 2 === 0) {
                median = (quantites[mid-1] + quantites[mid]) / 2;
            } else {
                median = quantites[mid];
            }
            
            // Mode (type le plus fréquent)
            const typeCount = {};
            data.forEach(d => {
                typeCount[d.type] = (typeCount[d.type] || 0) + 1;
            });
            let mode = null;
            let maxCount = 0;
            for (let [type, count] of Object.entries(typeCount)) {
                if (count > maxCount) {
                    maxCount = count;
                    mode = type;
                }
            }
            
            // Statistiques par plage
            const plageStats = {};
            data.forEach(d => {
                if (!plageStats[d.plage]) {
                    plageStats[d.plage] = { total: 0, count: 0 };
                }
                plageStats[d.plage].total += d.quantite;
                plageStats[d.plage].count += 1;
            });
            
            return {
                totalObservations: data.length,
                totalDechets: sum,
                moyenne: mean.toFixed(2),
                mediane: median.toFixed(2),
                min: Math.min(...quantites),
                max: Math.max(...quantites),
                mode: mode,
                plagePlusPolluee: Object.entries(plageStats).sort((a,b) => b[1].total - a[1].total)[0]?.[0] || 'Aucune',
                parType: typeCount,
                parPlage: plageStats
            };
        }
        
        function loadAndDisplayStats() {
            const data = getData();
            const stats = calculateStats(data);
            
            const container = document.getElementById('statsContainer');
            if (!stats) {
                container.innerHTML = '<p>Aucune donnée collectée pour le moment. Commencez par ajouter des données !</p>';
                return;
            }
            
            container.innerHTML = `
                <div class="stats-grid">
                    <div class="stat-card">
                        <h3>Total collectes</h3>
                        <div class="value">${stats.totalObservations}</div>
                    </div>
                    <div class="stat-card">
                        <h3>Total déchets (kg)</h3>
                        <div class="value">${stats.totalDechets} kg</div>
                    </div>
                    <div class="stat-card">
                        <h3>Moyenne par collecte</h3>
                        <div class="value">${stats.moyenne} kg</div>
                    </div>
                    <div class="stat-card">
                        <h3>Médiane</h3>
                        <div class="value">${stats.mediane} kg</div>
                    </div>
                    <div class="stat-card">
                        <h3>Minimum</h3>
                        <div class="value">${stats.min} kg</div>
                    </div>
                    <div class="stat-card">
                        <h3>Maximum</h3>
                        <div class="value">${stats.max} kg</div>
                    </div>
                    <div class="stat-card">
                        <h3>Déchet le plus fréquent</h3>
                        <div class="value">${stats.mode}</div>
                    </div>
                    <div class="stat-card">
                        <h3>Plage la plus polluée</h3>
                        <div class="value">${stats.plagePlusPolluee}</div>
                    </div>
                </div>
                
                <h3 style="margin-top: 30px;">📊 Répartition par type de déchet</h3>
                ${generateTypeChart(stats.parType)}
                
                <h3 style="margin-top: 30px;">🏖️ Statistiques par plage</h3>
                ${generatePlageTable(stats.parPlage)}
            `;
        }
        
        function generateTypeChart(parType) {
            let html = '<table><thead><tr><th>Type de déchet</th><th>Fréquence</th><th>Pourcentage</th></tr></thead><tbody>';
            const total = Object.values(parType).reduce((a,b) => a+b, 0);
            for (let [type, count] of Object.entries(parType)) {
                const percent = ((count / total) * 100).toFixed(1);
                html += `<tr>
                    <td>${type}</td>
                    <td>${count}</td>
                    <td>${percent}%</td>
                </tr>`;
            }
            html += '</tbody></table>';
            return html;
        }
        
        function generatePlageTable(parPlage) {
            let html = '<table><thead><tr><th>Plage</th><th>Nombre de collectes</th><th>Total déchets (kg)</th><th>Moyenne (kg)</th></tr></thead><tbody>';
            for (let [plage, stats] of Object.entries(parPlage)) {
                const moyenne = (stats.total / stats.count).toFixed(2);
                html += `<tr>
                    <td>${plage}</td>
                    <td>${stats.count}</td>
                    <td>${stats.total} kg</td>
                    <td>${moyenne} kg</td>
                </tr>`;
            }
            html += '</tbody></table>';
            return html;
        }
        
        function loadAndDisplayTable() {
            const data = getData();
            const container = document.getElementById('tableContainer');
            
            if (data.length === 0) {
                container.innerHTML = '<p>Aucune donnée collectée pour le moment.</p>';
                return;
            }
            
            // Trier par date (plus récent en premier)
            const sortedData = [...data].sort((a,b) => new Date(b.date) - new Date(a.date));
            
            let html = `
                <button onclick="exportToCSV()" style="margin-bottom: 15px; background: #28a745;">📥 Exporter en CSV</button>
                <button onclick="clearAllData()" style="margin-bottom: 15px; margin-left: 10px; background: #dc3545;">🗑️ Effacer toutes les données</button>
                <table>
                    <thead>
                        <tr>
                            <th>Date</th>
                            <th>Plage</th>
                            <th>Quantité (kg)</th>
                            <th>Type</th>
                            <th>Bénévoles</th>
                            <th>Action</th>
                        </tr>
                    </thead>
                    <tbody>
            `;
            
            sortedData.forEach(item => {
                html += `
                    <tr>
                        <td>${item.date}</td>
                        <td>${item.plage}</td>
                        <td>${item.quantite}</td>
                        <td>${item.type}</td>
                        <td>${item.benevoles}</td>
                        <td><button onclick="deleteItem(${item.id})" style="background:#dc3545; padding:5px 10px;">❌</button></td>
                    </tr>
                `;
            });
            
            html += '</tbody></table>';
            container.innerHTML = html;
        }
        
        function deleteItem(id) {
            let data = getData();
            data = data.filter(item => item.id !== id);
            saveData(data);
            loadAndDisplayStats();
            loadAndDisplayTable();
            showTab('donnees');
        }
        
        function clearAllData() {
            if (confirm('⚠️ Êtes-vous sûr de vouloir effacer TOUTES les données ? Cette action est irréversible.')) {
                saveData([]);
                loadAndDisplayStats();
                loadAndDisplayTable();
                showTab('donnees');
            }
        }
        
        function exportToCSV() {
            const data = getData();
            if (data.length === 0) {
                alert('Aucune donnée à exporter');
                return;
            }
            
            let csv = 'Date,Plage,Quantité (kg),Type de déchet,Nombre de bénévoles\n';
            data.forEach(item => {
                csv += `${item.date},${item.plage},${item.quantite},${item.type},${item.benevoles}\n`;
            });
            
            const blob = new Blob([csv], { type: 'text/csv' });
            const link = document.createElement('a');
            link.href = URL.createObjectURL(blob);
            link.download = `eco-ocean_data_${new Date().toISOString().split('T')[0]}.csv`;
            link.click();
        }
        
        function showTab(tabName) {
            // Cacher tous les contenus
            document.querySelectorAll('.tab-content').forEach(content => {
                content.classList.remove('active');
            });
            // Désactiver tous les boutons
            document.querySelectorAll('.tab').forEach(tab => {
                tab.classList.remove('active');
            });
            
            // Afficher l'onglet sélectionné
            document.getElementById(tabName).classList.add('active');
            event.target.classList.add('active');
            
            // Recharger les données si nécessaire
            if (tabName === 'analyse') {
                loadAndDisplayStats();
            } else if (tabName === 'donnees') {
                loadAndDisplayTable();
            }
        }
        
        // Rendre les fonctions globales
        window.showTab = showTab;
        window.deleteItem = deleteItem;
        window.clearAllData = clearAllData;
        window.exportToCSV = exportToCSV;
    </script>
